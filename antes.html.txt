// 3. Lógica de Selects (Estados y Municipios)
function populateEstados() {
    if (typeof estadosConMunicipios === 'undefined') return;
    const estados = Object.keys(estadosConMunicipios).sort();
    const selectMatriz = document.getElementById('estadoMatriz');
    const selectUbicacion = document.getElementById('estado-input');
    
    // Limpiar y poner opción default
    if(selectMatriz) selectMatriz.innerHTML = '<option value="" selected>-- Seleccione un Estado --</option>';
    if(selectUbicacion) selectUbicacion.innerHTML = '<option value="" selected>-- Estado --</option>';

    estados.forEach(estado => {
        const option = `<option value="${estado}">${estado}</option>`;
        if(selectMatriz) selectMatriz.insertAdjacentHTML('beforeend', option);
        if(selectUbicacion) selectUbicacion.insertAdjacentHTML('beforeend', option);
    });
}

function populateMunicipios(estado, municipioSelectId) {
    const selectMunicipio = document.getElementById(municipioSelectId);
    if(!selectMunicipio) return;

    selectMunicipio.innerHTML = '';
    
    if (estado && typeof estadosConMunicipios !== 'undefined' && estadosConMunicipios[estado]) {
        selectMunicipio.disabled = false;
        selectMunicipio.innerHTML = `<option value="">-- Seleccione un Municipio --</option>`;
        estadosConMunicipios[estado].sort().forEach(municipio => {
            selectMunicipio.insertAdjacentHTML('beforeend', `<option value="${municipio}">${municipio}</option>`);
        });
    } else {
        selectMunicipio.disabled = true;
        const defaultText = municipioSelectId === 'municipio-input' ? '-- Municipio --' : '-- Seleccione --';
        selectMunicipio.innerHTML = `<option value="">${defaultText}</option>`;
    }
}

// 4. (NUEVO) Lógica de Tabla Siniestros (Agregar/Quitar Filas)
function addSiniestroRow(tipo = '', num = '', monto = '') {
    const tbody = document.getElementById('siniestrosTableBody');
    if (!tbody) return;

    const row = document.createElement('tr');
    row.innerHTML = `
        <td>
            <input type="text" class="form-control form-control-sm siniestro-tipo" placeholder="Ej. Colisión" value="${tipo}">
        </td>
        <td>
            <input type="number" class="form-control form-control-sm siniestro-num text-center" placeholder="0" value="${num}">
        </td>
        <td>
            <div class="input-group input-group-sm">
                <span class="input-group-text">$</span>
                <input type="number" class="form-control siniestro-monto text-end" placeholder="0.00" value="${monto}">
            </div>
        </td>
        <td class="text-center align-middle">
            <span class="btn-remove-row" onclick="removeSiniestroRow(this)" title="Eliminar fila">&times;</span>
        </td>
    `;
    tbody.appendChild(row);
}

function removeSiniestroRow(btn) {
    const row = btn.closest('tr');
    const tbody = document.getElementById('siniestrosTableBody');
    // Si es la única fila, limpiar valores en vez de borrar para no dejar la tabla vacía
    if (tbody.children.length > 1) {
        row.remove();
    } else {
        row.querySelectorAll('input').forEach(input => input.value = '');
    }
}

    const ctx = document.getElementById('compliance-radar-chart').getContext('2d');
    if (typeof data === 'undefined') return;

        type: 'radar',
        data: {
            labels: Object.keys(data),
            datasets: [{
                label: 'Nivel de Cumplimiento',
                data: [],
                backgroundColor: 'rgba(130, 2, 124, 0.15)',
                borderColor: 'rgba(130, 2, 124, 1)',
                borderWidth: 3,
                pointBackgroundColor: function(context) {
                    const value = context.raw;
                    if (value >= 8) return '#20c997';
                    if (value >= 6) return '#ffc107';
                    if (value >= 3) return '#fd7e14';
                    return '#dc3545';
                },
                pointBorderColor: '#fff',
                pointBorderWidth: 2,
                pointRadius: 6,
                pointHoverRadius: 8,
                fill: true
            }, {
                label: 'Área de Riesgo Alto',
                data: Object.keys(data).map(() => 6),
                backgroundColor: 'rgba(255, 193, 7, 0.1)',
                borderColor: 'rgba(255, 193, 7, 0.5)',
                borderWidth: 1,
                borderDash: [5, 5],
                pointRadius: 0,
                fill: false
            }, {
                label: 'Área de Riesgo Extremo',
                data: Object.keys(data).map(() => 3),
                backgroundColor: 'rgba(220, 53, 69, 0.1)',
                borderColor: 'rgba(220, 53, 69, 0.5)',
                borderWidth: 1,
                borderDash: [3, 3],
                pointRadius: 0,
                fill: false
            }]
        },
        options: {
            scales: {
                r: {
                    angleLines: { color: 'rgba(0, 0, 0, 0.1)', lineWidth: 1 },
                    grid: { 
                        color: function(context) {
                            if (context.tick.value === 8) return 'rgba(32, 201, 151, 0.4)';
                            if (context.tick.value === 6) return 'rgba(255, 193, 7, 0.4)';
                            if (context.tick.value === 3) return 'rgba(220, 53, 69, 0.4)';
                            return 'rgba(0, 0, 0, 0.1)';
                        },
                        lineWidth: function(context) {
                            if ([3, 6, 8].includes(context.tick.value)) return 2;
                            return 1;
                        }
                    },
                    pointLabels: { 
                        font: { size: 11, weight: '600' },
                        color: '#2c3e50'
                    },
                    beginAtZero: true,
                    max: 10,
                    ticks: { 
                        stepSize: 2, 
                        backdropColor: 'rgba(255, 255, 255, 0.9)', 
                        color: '#666',
                        font: { size: 10 }
                    }
                }
            },
            maintainAspectRatio: true,
            plugins: { 
                legend: { display: false },
                tooltip: {
                    callbacks: {
                        label: function(context) {
                            const value = context.raw;
                            let riskLevel = 'EXTREMO';
                            if (value >= 8) riskLevel = 'BAJO';
                            else if (value >= 6) riskLevel = 'TOLERABLE';
                            else if (value >= 3) riskLevel = 'ALTO';
                            return `Puntuación: ${value.toFixed(1)} (${riskLevel})`;
                        }
                    }
                }
            }
        }
    });
}

function getRiskLevel(score) {
    if (score >= 8) return { text: 'BAJO', class: 'risk-badge-BAJO', classHtml: 'risk-bajo' };
    if (score >= 6) return { text: 'TOLERABLE', class: 'risk-badge-TOLERABLE', classHtml: 'risk-tolerable' };
    if (score >= 3) return { text: 'ALTO', class: 'risk-badge-ALTO', classHtml: 'risk-alto' };
    return { text: 'EXTREMO', class: 'risk-badge-EXTREMO', classHtml: 'risk-extremo' };
}

// --- FUNCIONES DE EVENTOS Y CONTROL DE FLUJO ---

function setupEventListeners() {
    // 1. Botones de Guardado
    const btnSave = document.getElementById('btnSaveForLater');
    const btnFinalize = document.getElementById('btnFinalize');
    
    if(btnSave) btnSave.addEventListener('click', () => saveSurvey(false));
    if(btnFinalize) btnFinalize.addEventListener('click', () => saveSurvey(true));

    // 2. Listeners para Selects de Estados (Cascada Municipios)
    const estadoMatriz = document.getElementById('estadoMatriz');
    if(estadoMatriz) {
        estadoMatriz.addEventListener('change', (e) => {
            populateMunicipios(e.target.value, 'municipioMatriz');
            checkFormCompletion();
        });
    }
    
    const estadoInput = document.getElementById('estado-input');
    if(estadoInput) {
        estadoInput.addEventListener('change', (e) => {
            populateMunicipios(e.target.value, 'municipio-input');
        });
    }

    // 3. Botones de Ubicaciones Adicionales
    const btnAddLoc = document.getElementById('agregar-ubicacion-btn');
    if(btnAddLoc) btnAddLoc.addEventListener('click', () => addLocation());
    
    const listaUbicaciones = document.getElementById('lista-ubicaciones');
    if(listaUbicaciones) {
        listaUbicaciones.addEventListener('click', (e) => {
            if (e.target.classList.contains('btn-close')) {
                e.target.closest('.location-item').remove();
                checkFormCompletion();
            }
        });
    }

    // 4. Listener unificado para botones "Retomar/Modificar" en AMBOS acordeones
    const handleResumeClick = (e) => {
        const resumeButton = e.target.closest('.resume-btn');
        if (resumeButton) {
            const surveyId = resumeButton.getAttribute('data-survey-id');
            const surveyData = window.allPendingSurveys.find(s => s.SurveyID === surveyId);

            if (surveyData) {
                populateForm(surveyData);
                
                // Cerrar acordeones de listas
                const bsCollapsePending = document.getElementById('collapsePending');
                const bsCollapseCompleted = document.getElementById('collapseCompleted');
                if(bsCollapsePending && bootstrap.Collapse.getInstance(bsCollapsePending)) bootstrap.Collapse.getInstance(bsCollapsePending).hide();
                if(bsCollapseCompleted && bootstrap.Collapse.getInstance(bsCollapseCompleted)) bootstrap.Collapse.getInstance(bsCollapseCompleted).hide();
                
                // Abrir primera sección
                const collapseOne = document.getElementById('collapseOne');
                if(collapseOne) new bootstrap.Collapse(collapseOne, {toggle: false}).show();

                window.scrollTo({ top: 0, behavior: 'smooth' });
            } else {
                alert('Error: Datos no encontrados en memoria local. Intente recargar la página.');
            }
        }
    };

    const pendingAcc = document.getElementById('pendingAccordion');
    const completedAcc = document.getElementById('completedAccordion');
    if(pendingAcc) pendingAcc.addEventListener('click', handleResumeClick);
    if(completedAcc) completedAcc.addEventListener('click', handleResumeClick);

    // 5. Animaciones y Control del Acordeón Principal
    const formAccordion = document.getElementById('formAccordion');
    if(formAccordion) {
        formAccordion.addEventListener('show.bs.collapse', event => {
            document.querySelectorAll('.nav-card').forEach(card => card.classList.remove('active'));
            const navCardTarget = document.querySelector(`.nav-card[data-bs-target="#${event.target.id}"]`);
            if (navCardTarget) navCardTarget.classList.add('active');
            
            const preguntas = event.target.querySelectorAll('.pregunta');
            anime.remove(preguntas);
            anime({
                targets: preguntas, opacity: [0, 1], translateY: [15, 0],
                delay: anime.stagger(40, { start: 100 }), duration: 400, easing: 'easeOutCubic'
            });
        });

        formAccordion.addEventListener('hide.bs.collapse', event => {
            const navCardTarget = document.querySelector(`.nav-card[data-bs-target="#${event.target.id}"]`);
            if (navCardTarget) navCardTarget.classList.remove('active');
        });
    }

    // 6. Listener Global de Cambios
    const mainForm = document.getElementById('main-form');
    if(mainForm) {
        mainForm.addEventListener('change', (e) => {
            updateProgress();
            checkFormCompletion();
        });
        
        mainForm.addEventListener('input', (e) => {
            const preguntaDiv = e.target.closest('.pregunta');
            if (preguntaDiv && e.target.value.trim() !== '') {
                preguntaDiv.classList.remove('unanswered');
            }
            checkFormCompletion();
        });
    }

    // 7. Navegación Lateral
    const navContainer = document.getElementById('nav-card-container');
    if(navContainer) {
        navContainer.addEventListener('click', (e) => {
            const navCard = e.target.closest('.nav-card');
            if (navCard) {
                const targetId = navCard.dataset.bsTarget;
                const targetElement = document.querySelector(targetId);
                if (targetElement) {
                    new bootstrap.Collapse(targetElement, { toggle: false }).show();
                }
            }
        });
    }

    // 8. Inicializar Sistema de Login
    initLoginSystem();
}

function initLoginSystem() {
    const loginOverlay = document.getElementById('login-overlay');
    const loginButton = document.getElementById('login-button');
    const mainContent = document.getElementById('main-content');
    const registerModal = document.getElementById('register-modal');
    
    const handleLogin = async () => {
        const user = document.getElementById('login-username').value.trim();
        const pass = document.getElementById('login-password').value.trim();
        const loginError = document.getElementById('login-error');
        
        if (!user || !pass) { loginError.textContent = 'Campos obligatorios.'; return; }
        
        loginButton.disabled = true; loginButton.textContent = 'Verificando...';
        
        try {
            const response = await fetch('login.php', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ username: user, password: pass })
            });
            const result = await response.json();
            
            if (result.success) {
                anime({ targets: loginOverlay, opacity: 0, duration: 500, easing: 'easeOutQuad', complete: () => { loginOverlay.style.display = 'none'; } });
                mainContent.style.display = 'block';
                anime({ targets: mainContent, opacity: [0, 1], duration: 800 });
                loadPendingSurveys();
            } else {
                loginError.textContent = result.message || 'Error de credenciales.';
                anime({ targets: '#login-box', translateX: [0, -10, 10, 0], duration: 400 });
                loginButton.disabled = false; loginButton.textContent = 'Ingresar';
            }
        } catch (error) {

            });

function checkFormCompletion() {
    const empresa = document.getElementById('empresaNombre');

function checkQuestionAnswered(q) {
    let answered = false;
    const preguntaDiv = document.querySelector(`.pregunta[data-question-id="${q.contador}"]`);
    if (!preguntaDiv) return true;

    if (q.informativa) {
        const el = document.getElementById(`respuesta_${q.contador}`);
        if (el && el.value.trim() !== '') answered = true;
    } else {
        const el = document.querySelector(`input[name="pregunta_${q.contador}"]:checked`);
        if (el) answered = true;
    }

    if (answered) preguntaDiv.classList.remove('unanswered');
    else preguntaDiv.classList.add('unanswered');
    return answered;
}

function updateProgress() {
    const activeSectionScores = []; 
    const chartLabels = []; 
    const chartData = []; 
    
    if(typeof data === 'undefined') return;

    Object.keys(data).forEach(area => {
        const questions = data[area];
        const areaId = area.replace(/\s+/g, '-');
        const gradable = questions.filter(q => !q.informativa);
        
        let correct = 0;
        let answered = 0;
        
        gradable.forEach(q => {
            const radio = document.querySelector(`input[name="pregunta_${q.contador}"]:checked`);
            if (radio) {
                answered++;
                const isInverse = [4, 58].includes(q.contador);
                if (isInverse ? radio.value === '0' : radio.value === '1') correct++;
            }
        });

        const score = gradable.length > 0 ? (correct / gradable.length) * 10 : 0;
        
        if (!currentDisabledSections.includes(area)) {
            activeSectionScores.push(score);
            chartLabels.push(area);
            chartData.push(score);
        }

        const risk = getRiskLevel(score);
        const badge = document.getElementById(`risk-level-${areaId}`);
        if(badge) {
            badge.textContent = risk.text;
            badge.className = 'risk-badge ' + risk.class;
        }

        const progressEl = document.getElementById(`progress-bar-${areaId}`);
        if(progressEl) progressEl.style.width = (gradable.length > 0 ? (answered/gradable.length)*100 : 0) + '%';
        
        const scoreEl = document.getElementById(`score-${areaId}`);
        if(scoreEl) scoreEl.textContent = score.toFixed(1);
        
        if(scoreCounters[areaId]) scoreCounters[areaId].value = score;
        document.getElementById(`answered-count-${areaId}`).textContent = `${answered} / ${gradable.length}`;
    });

    let generalScore = 0;
    if (activeSectionScores.length > 0) {
        generalScore = (activeSectionScores.reduce((a, b) => a + b, 0) / activeSectionScores.length) * 10;
    }
    
    const genScoreEl = document.getElementById('general-score-value');
    if(genScoreEl) genScoreEl.textContent = generalScore.toFixed(1) + ' / 100';
    
    const overallRisk = getRiskLevel(generalScore/10);
    const overallBadge = document.getElementById('overall-risk-level');
    if(overallBadge) {
        overallBadge.textContent = overallRisk.text;
        overallBadge.className = 'risk-badge ' + overallRisk.class;
    }

    if (complianceChart) {
        complianceChart.data.labels = chartLabels;
        complianceChart.data.datasets[0].data = chartData;
        // Actualizar las líneas de referencia de riesgo
        complianceChart.data.datasets[1].data = chartLabels.map(() => 6);
        complianceChart.data.datasets[2].data = chartLabels.map(() => 3);
        complianceChart.update();
    }
}

function toggleSection(buttonEl, sectionName) {
    const sectionCard = buttonEl.closest('.section-card'); 
    const index = currentDisabledSections.indexOf(sectionName);

    if (index > -1) {
        currentDisabledSections.splice(index, 1); // Activar
    } else {
        currentDisabledSections.push(sectionName); // Desactivar
    }
    updateSectionToggleUI();
    updateProgress(); 
    checkFormCompletion();
}

function updateSectionToggleUI() {
    document.querySelectorAll('.toggle-section-btn').forEach(buttonEl => {
        const onclickAttr = buttonEl.getAttribute('onclick');
        if(!onclickAttr) return;
        const sectionName = onclickAttr.split("'")[1]; 
        const sectionCard = buttonEl.closest('.section-card');

        if (currentDisabledSections.includes(sectionName)) {
            sectionCard.classList.add('disabled');
            buttonEl.textContent = 'Activar';
            buttonEl.classList.add('btn-success');
            buttonEl.classList.remove('btn-outline-secondary');
        } else {
            sectionCard.classList.remove('disabled');
            buttonEl.textContent = 'Desactivar';
            buttonEl.classList.remove('btn-outline-secondary');
            buttonEl.classList.add('btn-success'); // Verde para indicar que se puede desactivar? O mejor gris
            // Ajuste visual: Mejor volver a outline-secondary
            buttonEl.classList.add('btn-outline-secondary');
            buttonEl.classList.remove('btn-success');
        }
    });
}

function getAnalisisMercancia(tiposSeleccionados) {
    const tipos = tiposSeleccionados.map(t => t.toLowerCase());
    const hasMiscelaneos = tipos.some(t => t.includes("misceláneos") || t.includes("miscelaneos"));
    const hasElectronicos = tipos.some(t => t.includes("electrónicos") || t.includes("electronicos"));
    const hasMetales = tipos.some(t => t.includes("metales"));
    const hasAuto = tipos.some(t => t.includes("auto") || t.includes("partes"));
    const hasAlimentos = tipos.some(t => t.includes("alimentos"));

    let analisis = "";

    if (hasMiscelaneos || hasElectronicos || hasMetales || hasAuto || hasAlimentos) {
        analisis += `
        <p>En base al tipo de carga movilizada, se referencia que los productos más robados en el tercer trimestre 2025 fueron <strong>Alimentos & Bebidas (25%)</strong>, <strong>Auto & Partes (12%)</strong> y <strong>Misceláneos (10%)</strong>.</p>
        <p>Al comparar los cambios porcentuales en la distribución de los robos por tipo de producto con relación al tercer trimestre 2024 se detectó que los incrementos más significativos correspondieron a las categorías Misceláneos, Electrónicos y Metales.</p>`;

        if (hasMiscelaneos) {
            analisis += `<p>• <strong>Misceláneos:</strong> La categoría aumentó cuatro puntos porcentuales, siendo el principal objetivo de los delincuentes las cargas mixtas como paquetería o mercancía de venta en línea.</p>`;
        }
        if (hasElectronicos) {
            analisis += `<p>• <strong>Electrónicos:</strong> Incrementó dos puntos porcentuales, siendo equipos de cómputo, televisores, celulares y consolas de juegos los más robados.</p>`;
        }
        if (hasMetales) {
            analisis += `<p>• <strong>Metales:</strong> Aumentó dos puntos porcentuales, siendo de los productos más robados acero, aluminio y cobre.</p>`;
        }
    } else {
        analisis = `<p>Los tipos de carga movilizadas presentan un riesgo de robo aproximado del 1% al 5%, lo cual no representa una desviación crítica, aunque se recomienda mantener protocolos estándar.</p>`;
    }
    return analisis;
}

function collectAllData() {
    // 1. Mercancía
    const mercanciaSeleccionada = [];
    document.querySelectorAll('input[name="tipoMercancia"]:checked').forEach(c => mercanciaSeleccionada.push(c.value));
    const otroMerc = document.getElementById('tipoMercanciaOtro').value.trim();
    if (otroMerc) mercanciaSeleccionada.push(`Otro: ${otroMerc}`);

    // 2. Estados Servicio
    const estadosServicio = [];
    document.querySelectorAll('input[name="estadosServicio"]:checked').forEach(c => estadosServicio.push(c.value));

    // 3. Tabla Siniestros
    const tablaSiniestros = [];
    document.querySelectorAll('#siniestrosTableBody tr').forEach(row => {
        const inputs = row.querySelectorAll('input');
        if(inputs.length >= 3) {
            const tipo = inputs[0].value.trim();
            const num = inputs[1].value.trim();
            const monto = inputs[2].value.trim();
            if(tipo || num || monto) tablaSiniestros.push({ tipo, num, monto });
        }
    });

    const datos = {
        empresa: document.getElementById('empresaNombre').value.trim(),
        descripcionEmpresa: document.getElementById('descripcionEmpresa').value.trim(), 
        fechaEntrevista: document.getElementById('fechaEntrevista').value.trim(),
        certificaciones: document.getElementById('certificacion').value.trim(),
        principalesClientes: document.getElementById('principalesClientes').value.trim(),
        tipoMercancia: mercanciaSeleccionada.join(', '),
        tipoMercanciaArray: mercanciaSeleccionada,
        
        estadosServicio: estadosServicio.join(', '), 
        estadosServicioArray: estadosServicio,
        
        numConductores: document.getElementById('numConductores').value,
        numTalleres: document.getElementById('numTalleres').value,
        kmRecorridosMes: document.getElementById('kmRecorridosMes').value,
        viajesAnio: document.getElementById('viajesAnio').value,
        lesionadosMes: document.getElementById('lesionadosMes').value,
        lesionadosAnio: document.getElementById('lesionadosAnio').value,
        muertosMes: document.getElementById('muertosMes').value,
        muertosAnio: document.getElementById('muertosAnio').value,

        estadoMatriz: document.getElementById('estadoMatriz').value,
        municipioMatriz: document.getElementById('municipioMatriz').value,
        ubicacionesAdicionales: Array.from(document.querySelectorAll('.location-item')).map(item => `${item.dataset.sucursal} (${item.dataset.municipio}, ${item.dataset.estado})`).join(' | '),

        // Siniestralidad
        fechaConsultaSiniestralidad: document.getElementById('fechaConsultaSiniestralidad').value,
        frecuenciaSiniestralidad: document.getElementById('frecuenciaSiniestralidad').value,
        severidadSiniestralidad: document.getElementById('severidadSiniestralidad').value,
        tablaSiniestros: tablaSiniestros,

        respuestasCuestionario: [],
        seccionesInfo: {},
        disabledSections: currentDisabledSections 
    };

    // Recoger Entrevistados
    if(typeof data !== 'undefined') {
        Object.keys(data).forEach(area => {
            const areaId = area.replace(/\s+/g, '-');
            datos.seccionesInfo[area] = {
                entrevistado: document.getElementById(`entrevistado_${areaId}`)?.value.trim() || '',
                puesto: document.getElementById(`puesto_${areaId}`)?.value.trim() || '',
                antiguedad: document.getElementById(`antiguedad_${areaId}`)?.value.trim() || ''
            };
        });

        // Recoger Preguntas
        Object.values(data).flat().forEach(q => {
            let resp = '';
            if (q.informativa) {
                const el = document.getElementById(`respuesta_${q.contador}`);
                if(el) resp = el.value.trim();
            } else {
                const el = document.querySelector(`input[name="pregunta_${q.contador}"]:checked`);
                if(el) resp = el.value;
            }
            const recEl = document.getElementById(`recomendacion_${q.contador}`);
            const rec = recEl ? recEl.value.trim() : '';
            
            datos.respuestasCuestionario.push({
                contador: q.contador, pregunta: q.pregunta, respuesta: resp,
                recomendacion: rec || q.recomendacion || ''
            });
        });
    }
    return datos;
}

function addLocation(sucursal, estado, municipio) {
    const sInput = document.getElementById('sucursal-input');
    const eInput = document.getElementById('estado-input');
    const mInput = document.getElementById('municipio-input');
    
    const sVal = sucursal || sInput.value.trim();
    const eVal = estado || eInput.value;
    const mVal = municipio || mInput.value;
    
    if (sVal && eVal && mVal) {
        const div = document.createElement('div');
        div.className = 'location-item d-flex justify-content-between align-items-center p-2 border rounded mb-2';
        div.dataset.sucursal = sVal;
        div.dataset.estado = eVal;
        div.dataset.municipio = mVal;
        div.innerHTML = `<span><strong>${sVal}</strong> - <em>${municipio}, ${eVal}</em></span><button type="button" class="btn-close"></button>`;
        document.getElementById('lista-ubicaciones').appendChild(div);
        
        if (!sucursal) {
            sInput.value = ''; eInput.value = ''; mInput.innerHTML = '<option value="">-- Municipio --</option>'; mInput.disabled = true;
        }
    } else if (!sucursal) {
        alert('Complete todos los campos.');
    }
}

async function saveSurvey(isFinalizing) {
    const btn = isFinalizing ? document.getElementById('btnFinalize') : document.getElementById('btnSaveForLater');
    const txt = btn.innerHTML;
    btn.disabled = true; btn.innerHTML = 'Guardando...';

    try {
        const payload = collectAllData();
        payload.status = isFinalizing ? 'completado' : 'pendiente';
        payload.surveyId = window.currentSurveyId || null;

        const response = await fetch('save_survey.php', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify(payload)
        });
        const res = await response.json();

        if (res.success) {
            window.currentSurveyId = res.surveyId;
            if (isFinalizing) {
                const scores = {};
                Object.keys(scoreCounters).forEach(k => scores[k.replace(/-/g,' ')] = scoreCounters[k].value);
                const generalScore = parseFloat(document.getElementById('general-score-value').textContent)||0;
                
                generateReport(payload, scores, generalScore);
                alert('Guardado y generando reporte...');
                loadPendingSurveys();
            } else {
                alert('Guardado correctamente.');
                loadPendingSurveys();
            }
        } else {
            alert('Error: ' + res.message);
        }
    } catch (e) {
        console.error(e);
        // Fallback offline
        if(isFinalizing) generateReport(collectAllData(), {}, 0);
    } finally {
        btn.disabled = false; btn.innerHTML = txt;
    }
}

function populateForm(d) {
    document.getElementById('main-form').reset();
    document.getElementById('lista-ubicaciones').innerHTML = '';
    window.currentSurveyId = d.SurveyID;

    // Campos directos
    safeSetValue('empresaNombre', d.Empresa);
    safeSetValue('descripcionEmpresa', d.descripcionEmpresa);
    
    // Fix Fecha ISO
    if(d.FechaEntrevista) safeSetValue('fechaEntrevista', d.FechaEntrevista.split('T')[0]);
    
    safeSetValue('certificacion', d.Certificaciones);
    safeSetValue('principalesClientes', d.PrincipalesClientes);
    safeSetValue('numeroPoliza', d.numeroPoliza);
    
    // Checkboxes
    if(d.TipoMercancia) {
        d.TipoMercancia.split(', ').forEach(t => {
            if(t.startsWith('Otro:')) safeSetValue('tipoMercanciaOtro', t.split(':')[1].trim());
            else {
                const cb = document.querySelector(`input[name="tipoMercancia"][value="${t}"]`);
                if(cb) cb.checked = true;
            }
        });
    }
    if(d.estadosServicio) {
        d.estadosServicio.split(', ').forEach(s => {
            const cb = document.querySelector(`input[name="estadosServicio"][value="${s}"]`);
            if(cb) cb.checked = true;
        });
    }

    // Sección 2
    safeSetValue('estadoMatriz', d.estadoMatriz);
    if(d.estadoMatriz) {
        populateMunicipios(d.estadoMatriz, 'municipioMatriz');
        safeSetValue('municipioMatriz', d.municipioMatriz);
    }
    safeSetValue('numConductores', d.numConductores);
    // ... otros campos numéricos ...
    safeSetValue('numTalleres', d.numTalleres);
    safeSetValue('kmRecorridosMes', d.kmRecorridosMes);
    safeSetValue('viajesAnio', d.viajesAnio);
    safeSetValue('lesionadosMes', d.lesionadosMes);
    safeSetValue('lesionadosAnio', d.lesionadosAnio);
    safeSetValue('muertosMes', d.muertosMes);
    safeSetValue('muertosAnio', d.muertosAnio);

    // Siniestralidad
    if(d.fechaConsultaSiniestralidad) safeSetValue('fechaConsultaSiniestralidad', d.fechaConsultaSiniestralidad.split('T')[0]);
    safeSetValue('frecuenciaSiniestralidad', d.frecuenciaSiniestralidad);
    safeSetValue('severidadSiniestralidad', d.severidadSiniestralidad);

    const tbody = document.getElementById('siniestrosTableBody');
    tbody.innerHTML = '';
    if(d.tablaSiniestros && Array.isArray(d.tablaSiniestros)) {
        d.tablaSiniestros.forEach(r => addSiniestroRow(r.tipo, r.num, r.monto));
    } else {
        addSiniestroRow();
    }

    // RESTAURAR SECCIONES DESACTIVADAS (Fix visual)
    if(d.SeccionesDesactivadas) {
        currentDisabledSections = d.SeccionesDesactivadas.split(',');
    } else {
        currentDisabledSections = [];
    }
    updateSectionToggleUI(); // Refrescar botones

    // Entrevistados y Preguntas
    if(typeof data !== 'undefined') {
        Object.keys(data).forEach(area => {
            safeSetValue(`entrevistado_${area.replace(/\s+/g,'-')}`, d['Entrevistado_'+area.replace(/ /g,'_')] || d['Entrevistado_'+area.replace(/[^a-zA-Z0-9]/g,'')]);
            safeSetValue(`puesto_${area.replace(/\s+/g,'-')}`, d['Puesto_'+area.replace(/ /g,'_')] || d['Puesto_'+area.replace(/[^a-zA-Z0-9]/g,'')]);
            safeSetValue(`antiguedad_${area.replace(/\s+/g,'-')}`, d['Antiguedad_'+area.replace(/ /g,'_')] || d['Antiguedad_'+area.replace(/[^a-zA-Z0-9]/g,'')]);
        });

        Object.values(data).flat().forEach(q => {
            const rKey = 'Pregunta_'+q.contador;
            const recKey = 'Recomendacion_'+q.contador;
            if(d[rKey]) {
                if(q.informativa) safeSetValue('respuesta_'+q.contador, d[rKey]);
                else {
                    const rad = document.querySelector(`input[name="pregunta_${q.contador}"][value="${d[rKey]}"]`);
                    if(rad) rad.checked = true;
                }
            }
            if(d[recKey]) safeSetValue('recomendacion_'+q.contador, d[recKey]);
            checkQuestionAnswered(q);
        });
    }
    
    updateProgress();
    checkFormCompletion();
    alert('Datos cargados y secciones restauradas.');
}

async function loadPendingSurveys() {
    const pContainer = document.getElementById('pending-surveys-list');
    const cContainer = document.getElementById('completed-surveys-list');
    
    const renderItem = (s, type) => {
        const date = s.Timestamp ? new Date(s.Timestamp).toLocaleDateString('es-MX') : 'S/F';
        const color = type === 'pendiente' ? 'warning' : 'success';
        return `
        <div class="card mb-2 border-${color}" style="border-left-width: 4px;">
            <div class="card-body p-2 d-flex justify-content-between align-items-center">
                <div><strong class="text-dark small">${s.Empresa || 'Sin Nombre'}</strong><div class="text-muted" style="font-size: 0.75rem;">ID: ${s.SurveyID} | ${date}</div></div>
                <button class="btn btn-sm btn-outline-${color} resume-btn" data-survey-id="${s.SurveyID}">${type === 'pendiente' ? 'Retomar' : 'Modificar'}</button>
            </div>
        </div>`;
    };

    try {
        const response = await fetch(`get_pending_surveys.php?t=${new Date().getTime()}`);
        if (!response.ok) throw new Error("Offline");
        const surveys = await response.json();
        window.allPendingSurveys = surveys;
        
        const pending = surveys.filter(s => s.Status === 'pendiente');
        const completed = surveys.filter(s => s.Status === 'completado');

        if (pContainer) pContainer.innerHTML = pending.length ? pending.map(s => renderItem(s, 'pendiente')).join('') : '<p class="text-muted small">No hay pendientes.</p>';
        if (cContainer) cContainer.innerHTML = completed.length ? completed.map(s => renderItem(s, 'completado')).join('') : '<p class="text-muted small">No hay historial.</p>';
    } catch (e) {
        if(pContainer) pContainer.innerHTML = '<p class="text-muted">Sin conexión.</p>';
    }
}
